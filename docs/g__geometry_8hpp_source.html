84constexpr GPolarDirection3D cartesianToPolarDirection(const GCartesianVector3D &cartesianDirection) {
85 const double azimuth = std::atan2(cartesianDirection.y, cartesianDirection.x);
86 const double magnitude =
87 std::sqrt(cartesianDirection.x * cartesianDirection.x + cartesianDirection.y * cartesianDirection.y +
88 cartesianDirection.z * cartesianDirection.z);
89 double elevation = std::asin(cartesianDirection.z / magnitude);
